# Angular SDK Guide

The RajutechieStreamKit Angular SDK (`@rajutechie-streamkit/angular-sdk`) provides Angular services, components, and pipes for integrating real-time chat, voice/video calls, meetings, and live streaming into Angular applications.

## Installation

```bash
npm install @rajutechie-streamkit/angular-sdk
# or
yarn add @rajutechie-streamkit/angular-sdk
# or
pnpm add @rajutechie-streamkit/angular-sdk
```

> **Note:** The Angular SDK has a peer dependency on `@angular/core >= 16.0.0` and `@rajutechie-streamkit/core`. Both are installed automatically.

## Module Setup

Import `RajutechieStreamKitModule` in your root `AppModule` using `forRoot()` to register all RajutechieStreamKit services as singletons:

```typescript
import { NgModule } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { RajutechieStreamKitModule } from '@rajutechie-streamkit/angular-sdk';
import { AppComponent } from './app.component';

@NgModule({
  declarations: [AppComponent],
  imports: [
    BrowserModule,
    RajutechieStreamKitModule.forRoot({
      apiKey: 'sk_live_xxxxx',
      region: 'us-east-1',
    }),
  ],
  bootstrap: [AppComponent],
})
export class AppModule {}
```

### Configuration Options

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `apiKey` | `string` | **required** | Your RajutechieStreamKit project API key |
| `region` | `string` | `'us-east-1'` | Deployment region |
| `apiUrl` | `string` | `'https://api.rajutechie-streamkit.io/v1'` | REST API base URL |
| `wsUrl` | `string` | `'wss://ws.rajutechie-streamkit.io'` | WebSocket server URL |
| `logLevel` | `LogLevel` | `LogLevel.WARN` | Logging verbosity |
| `autoReconnect` | `boolean` | `true` | Automatically reconnect on connection loss |

## Connecting

Inject `RajutechieStreamKitService` and call `connect()` with a user token (JWT generated by your server):

```typescript
import { Component, OnInit, OnDestroy } from '@angular/core';
import { RajutechieStreamKitService } from '@rajutechie-streamkit/angular-sdk';
import { Subscription } from 'rxjs';

@Component({
  selector: 'app-root',
  template: `<router-outlet *ngIf="isConnected"></router-outlet>`,
})
export class AppComponent implements OnInit, OnDestroy {
  isConnected = false;
  private sub?: Subscription;

  constructor(private rajutechie-streamkit: RajutechieStreamKitService) {}

  async ngOnInit() {
    const userToken = await this.fetchTokenFromServer();
    await this.rajutechie-streamkit.connect(userToken);

    this.sub = this.rajutechie-streamkit.connectionState$.subscribe((state) => {
      this.isConnected = state === 'connected';
      console.log('Connection state:', state);
    });
  }

  ngOnDestroy() {
    this.sub?.unsubscribe();
    this.rajutechie-streamkit.disconnect();
  }

  private async fetchTokenFromServer(): Promise<string> {
    const res = await fetch('/api/rajutechie-streamkit/token');
    const data = await res.json();
    return data.token;
  }
}
```

### RajutechieStreamKitService API

| Member | Type | Description |
|--------|------|-------------|
| `client` | `RajutechieStreamKitClient` | The underlying core client instance |
| `connectionState$` | `Observable<ConnectionState>` | Reactive connection state stream |
| `connectionState` | `ConnectionState` | Current connection state snapshot |
| `isConnected` | `boolean` | Whether the client is currently connected |
| `connect(token)` | `Promise<void>` | Connect with a user JWT |
| `disconnect()` | `Promise<void>` | Disconnect and clean up |

## RajutechieStreamKitChatService

Inject `RajutechieStreamKitChatService` for all chat operations. Methods return RxJS Observables, following Angular conventions.

### Fetching and Sending Messages

```typescript
import { Component, OnInit } from '@angular/core';
import { RajutechieStreamKitChatService } from '@rajutechie-streamkit/angular-sdk';
import type { Message } from '@rajutechie-streamkit/core';

@Component({
  selector: 'app-chat',
  template: `
    <div *ngFor="let msg of messages">
      <strong>{{ msg.senderId }}</strong>: {{ msg.content.text }}
    </div>
    <input #input (keydown.enter)="send(input.value); input.value = ''" />
  `,
})
export class ChatComponent implements OnInit {
  messages: Message[] = [];

  constructor(private chatService: RajutechieStreamKitChatService) {}

  ngOnInit() {
    this.chatService.getMessages('channel-id').subscribe((messages) => {
      this.messages = messages;
    });
  }

  send(text: string) {
    this.chatService
      .sendMessage('channel-id', { text })
      .subscribe((message) => {
        this.messages = [...this.messages, message];
      });
  }
}
```

### Real-Time Message Stream

```typescript
// Listen for new messages in real time
this.chatService.messageStream('channel-id').subscribe((message) => {
  this.messages = [...this.messages, message];
});

// Listen for message edits
this.chatService.messageUpdates('channel-id').subscribe((updated) => {
  this.messages = this.messages.map((m) => (m.id === updated.id ? updated : m));
});

// Listen for message deletions
this.chatService.messageDeletions('channel-id').subscribe(({ messageId }) => {
  this.messages = this.messages.filter((m) => m.id !== messageId);
});
```

### Typing Indicators

```typescript
// Signal that the current user is typing
this.chatService.startTyping('channel-id');

// Signal that the current user stopped typing
this.chatService.stopTyping('channel-id');

// Listen for typing events from other users
this.chatService.typingStream('channel-id').subscribe((event) => {
  console.log(`${event.userId} is typing in ${event.channelId}`);
});
```

### Reactions and Read Receipts

```typescript
// Add a reaction
this.chatService.addReaction('channel-id', 'message-id', '👍').subscribe();

// Remove a reaction
this.chatService.removeReaction('channel-id', 'message-id', '👍').subscribe();

// Mark a message as read
this.chatService.markAsRead('channel-id', 'message-id').subscribe();
```

## RajutechieStreamKitCallService

Inject `RajutechieStreamKitCallService` for voice and video call operations.

### Making and Receiving Calls

```typescript
import { Component, OnInit, OnDestroy } from '@angular/core';
import { RajutechieStreamKitCallService } from '@rajutechie-streamkit/angular-sdk';
import { Subscription } from 'rxjs';
import type { Call } from '@rajutechie-streamkit/core';

@Component({
  selector: 'app-call',
  template: `
    <div *ngIf="incomingCall">
      <p>Incoming call from {{ incomingCall.initiatedBy }}</p>
      <button (click)="accept()">Accept</button>
      <button (click)="reject()">Reject</button>
    </div>
    <button (click)="startVideoCall()">Start Video Call</button>
  `,
})
export class CallComponent implements OnInit, OnDestroy {
  incomingCall: Call | null = null;
  private subs: Subscription[] = [];

  constructor(private callService: RajutechieStreamKitCallService) {}

  ngOnInit() {
    this.subs.push(
      this.callService.incomingCalls$.subscribe((call) => {
        this.incomingCall = call;
      })
    );
  }

  startVideoCall() {
    this.callService
      .startCall({ type: 'video', participants: ['user-2'] })
      .subscribe((call) => {
        console.log('Call started:', call.id);
      });
  }

  accept() {
    if (this.incomingCall) {
      this.callService.acceptCall(this.incomingCall.id).subscribe();
    }
  }

  reject() {
    if (this.incomingCall) {
      this.callService.rejectCall(this.incomingCall.id, 'Busy').subscribe();
      this.incomingCall = null;
    }
  }

  ngOnDestroy() {
    this.subs.forEach((s) => s.unsubscribe());
  }
}
```

### Media Controls

```typescript
// Toggle audio (mute/unmute)
this.callService.toggleAudio(callId, false);  // mute
this.callService.toggleAudio(callId, true);   // unmute

// Toggle video (camera on/off)
this.callService.toggleVideo(callId, false);  // camera off
this.callService.toggleVideo(callId, true);   // camera on

// Switch front/back camera
this.callService.switchCamera(callId);

// Screen sharing
this.callService.startScreenShare(callId);
this.callService.stopScreenShare(callId);

// Recording
this.callService.startRecording(callId).subscribe();
this.callService.stopRecording(callId).subscribe();
```

### Call Event Observables

| Observable | Type | Description |
|------------|------|-------------|
| `incomingCalls$` | `Observable<Call>` | Emits on each incoming call |
| `callAccepted$` | `Observable<{ callId, userId }>` | Emits when a call is accepted |
| `callRejected$` | `Observable<{ callId, userId, reason? }>` | Emits when a call is rejected |
| `callEnded$` | `Observable<{ callId, reason }>` | Emits when a call ends |
| `participantJoined$` | `Observable<CallParticipant>` | Emits when a participant joins |
| `participantLeft$` | `Observable<{ callId, userId }>` | Emits when a participant leaves |

## RajutechieStreamKitMeetingService

Inject `RajutechieStreamKitMeetingService` for scheduling, joining, and managing meetings.

### Scheduling and Joining

```typescript
import { RajutechieStreamKitMeetingService } from '@rajutechie-streamkit/angular-sdk';

// Schedule a meeting
this.meetingService.scheduleMeeting({
  title: 'Sprint Planning',
  scheduledAt: '2026-03-01T10:00:00Z',
  duration: 60,
  participants: ['user-1', 'user-2'],
  settings: { muteOnJoin: true, waitingRoom: true },
}).subscribe((meeting) => {
  console.log('Meeting scheduled:', meeting.meetingCode);
});

// Join a meeting
this.meetingService.joinMeeting('meeting-id').subscribe((participant) => {
  console.log('Joined as:', participant.role);
});

// Join by shareable code
this.meetingService.joinByCode('ABC-DEF-GHI').subscribe((result) => {
  console.log('Joined meeting:', result.title);
});

// Leave a meeting
this.meetingService.leaveMeeting('meeting-id').subscribe();

// End a meeting (host only)
this.meetingService.endMeeting('meeting-id').subscribe();
```

### Participant Management

```typescript
// Observe participants in real time
this.meetingService.participants$('meeting-id').subscribe((participants) => {
  this.participantList = participants;
});

// Mute a participant (host only)
this.meetingService.muteParticipant('meeting-id', 'user-id').subscribe();

// Remove a participant (host only)
this.meetingService.removeParticipant('meeting-id', 'user-id').subscribe();

// Mute all (host only)
this.meetingService.muteAll('meeting-id').subscribe();
```

### Hand Raise and Polls

```typescript
// Raise / lower hand
this.meetingService.raiseHand('meeting-id');
this.meetingService.lowerHand('meeting-id');

// Listen for hand raise events
this.meetingService.handRaised$.subscribe(({ meetingId, userId }) => {
  console.log(`${userId} raised hand in ${meetingId}`);
});

// Create a poll
this.meetingService.createPoll('meeting-id', {
  question: 'Which date works best?',
  options: [
    { id: 'opt-1', text: 'Monday' },
    { id: 'opt-2', text: 'Wednesday' },
  ],
  anonymous: false,
}).subscribe((poll) => {
  console.log('Poll created:', poll.id);
});

// Vote on a poll
this.meetingService.votePoll('meeting-id', 'poll-id', 'opt-2').subscribe();

// Listen for new polls and results
this.meetingService.pollCreated$.subscribe((poll) => { /* ... */ });
this.meetingService.pollResult$.subscribe((poll) => { /* ... */ });
```

### Breakout Rooms

```typescript
this.meetingService.createBreakoutRooms('meeting-id', {
  count: 3,
  assignAutomatically: true,
}).subscribe((rooms) => {
  console.log('Created', rooms.length, 'breakout rooms');
```

## Pipes

The Angular SDK includes custom pipes for common formatting tasks:

```html

{{ message.createdAt | skTimeAgo }}     <!-- "2 minutes ago" -->

<!-- User presence status (returns Observable, use with async pipe) -->
{{ userId | skUserStatus | async }}      <!-- "online" / "offline" -->
```

## Full Chat Example

A complete chat component with channel list, message display, and input.

```typescript
import { Component, OnInit, OnDestroy } from '@angular/core';
import {
  RajutechieStreamKitService,
  RajutechieStreamKitChatService,
} from '@rajutechie-streamkit/angular-sdk';
import type { Message, TypingEvent } from '@rajutechie-streamkit/core';
import { Subscription } from 'rxjs';

@Component({
  selector: 'app-chat-room',
  template: `
    <div class="chat-layout">
      <!-- Channel list -->
      <sk-channel-list
        (channelSelected)="selectChannel($event.id)"
      ></sk-channel-list>

      <!-- Chat area -->
      <div class="chat-area" *ngIf="selectedChannel">
        <div class="messages">
          <div *ngFor="let msg of messages" class="message"
               [class.own]="msg.senderId === currentUserId">
            <strong>{{ msg.senderId }}</strong>
            <p>{{ msg.content.text }}</p>
            <small>{{ msg.createdAt | skTimeAgo }}</small>
          </div>
        </div>

        <div class="typing" *ngIf="typingUsers.length > 0">
          {{ typingUsers.join(', ') }}
          {{ typingUsers.length === 1 ? 'is' : 'are' }} typing...
        </div>

        <div class="input-row">
          <input
            #msgInput
            placeholder="Type a message..."
            (keydown.enter)="send(msgInput.value); msgInput.value = ''"
            (input)="onTyping()"
          />
          <button (click)="send(msgInput.value); msgInput.value = ''">
            Send
          </button>
        </div>
      </div>
    </div>
  `,
})
export class ChatRoomComponent implements OnInit, OnDestroy {
  selectedChannel: string | null = null;
  messages: Message[] = [];
  typingUsers: string[] = [];
  currentUserId = 'current-user';

  private subs: Subscription[] = [];

  constructor(
    private rajutechie-streamkit: RajutechieStreamKitService,
    private chatService: RajutechieStreamKitChatService,
  ) {}

  async ngOnInit() {
    const token = await this.fetchToken();
    await this.rajutechie-streamkit.connect(token);
  }

  selectChannel(channelId: string) {
    // Clean up previous subscriptions
    this.subs.forEach((s) => s.unsubscribe());
    this.subs = [];

    this.selectedChannel = channelId;

    // Load existing messages
    this.subs.push(
      this.chatService.getMessages(channelId).subscribe((messages) => {
        this.messages = messages;
      })
    );

    // Listen for new messages
    this.subs.push(
      this.chatService.messageStream(channelId).subscribe((message) => {
        this.messages = [...this.messages, message];
      })
    );

    // Listen for typing indicators
    this.subs.push(
      this.chatService.typingStream(channelId).subscribe((event) => {
        if (event.userId !== this.currentUserId) {
          this.typingUsers = [...new Set([...this.typingUsers, event.userId])];
          setTimeout(() => {
            this.typingUsers = this.typingUsers.filter((u) => u !== event.userId);
          }, 3000);
        }
      })
    );
  }

  send(text: string) {
    if (!text.trim() || !this.selectedChannel) return;
    this.chatService
      .sendMessage(this.selectedChannel, { text })
      .subscribe();
    this.chatService.stopTyping(this.selectedChannel);
  }

  onTyping() {
    if (this.selectedChannel) {
      this.chatService.startTyping(this.selectedChannel);
    }
  }

  ngOnDestroy() {
    this.subs.forEach((s) => s.unsubscribe());
    this.rajutechie-streamkit.disconnect();
  }

  private async fetchToken(): Promise<string> {
    const res = await fetch('/api/rajutechie-streamkit/token');
    const data = await res.json();
    return data.token;
  }
}
```

## Tips

- **Call `forRoot()` only once** in your root `AppModule`. All RajutechieStreamKit services are provided as singletons.
- **Cleanup is automatic.** Services implement `OnDestroy` and tear down event listeners and WebSocket subscriptions when the module is destroyed.
- **Use the `async` pipe** with the reactive Observables (`connectionState$`, `participants$`, `skUserStatus`) to avoid manual subscription management.
- **Token refresh:** Listen to `connectionState$` for `'disconnected'` events and re-authenticate with a fresh token from your server.
